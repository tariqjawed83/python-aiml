def applyGaussianBlurring(imgData):
  for i in range(0, len(imgData), 1):
    b = cv2.GaussianBlur(imgData[i], ksize=(7,7), sigmaX=0, sigmaY=0) # When sigmaX and sigmaY both are zero, values will be computed based on gaussian kernel size
    imgData[i] = np.asarray(b)

def printClasswiseConfusionMatrix(classIndex):
  tn, fp, fn, tp = multi_confusion[classIndex].ravel()
  print("True Negatives : ", tn)
  print("False Positives : ", fp)
  print("False Negatives : ", fn)
  print("True Positives : ", tp)

  plt.figure(figsize=(10,5))
  plt.title("Confusion metrics for class " + classList[classIndex])
  sns.heatmap(multi_confusion[classIndex], annot=True, fmt='', cmap="YlGnBu", xticklabels = ['Predicted 0','Predicted 1'] , yticklabels = ['Actual 0','Actual 1']);

def evaluateSpecificPrediction(test_item_index):
  Y_Predict_Temp = model.predict(X_Test[test_item_index].reshape(1, 128, 128, 3))
  Y_Predict_Temp[0] = Y_Predict_Temp[0] >= Y_Predict_Temp[0].max()
  Y_Predict_Temp = Y_Predict_Temp.astype('int')
  classIndexPredict = np.where(Y_Predict_Temp.flatten() == 1)[0][0]
  classIndexActual = np.where(Y_Test[test_item_index].flatten() == 1)[0][0]
  if (classIndexPredict == classIndexPredict):
    print("Actual class and Prediction class matched i.e. ", classList[classIndexPredict])
  else:
    print("Actual class ", classList[classIndexActual], " and Prediction class ", classList[classIndexPredict], " are different.")
